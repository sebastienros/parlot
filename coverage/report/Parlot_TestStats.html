 717    /// <summary>Should cover the case with the `expected` to be an array as well.</summary>
 718    public bool AreEqual<T>(IEnumerable<T> expected, IEnumerable<T> actual,
 719        [CallerArgumentExpression(nameof(expected))] string expectedName = "<expected>",
 720        [CallerArgumentExpression(nameof(actual))] string actualName = "<actual>",
 721        [CallerMemberName] string testName = "<test>", [CallerLineNumber] int sourceLineNumber = -1)

 724        var actualEnumerator = actual.GetEnumerator();
 725        var expectedCount = 0;
 726        var actualCount = 0;

 728        // Collecting the context around the non-equal items, lets call it a window,
 729        // If the window size is 8 it means:
 730        // - it will be 4 or less equal items before the first non-equal,
 731        // - 8 or less equal items between the non-equal items (if there more than 8 equal items in between, then it wil
 732        // - 4 or less equal items after the last non-equal item
 733

 735        // #0  == 1, 1
 736        // #1  == 3, 3
 737        // #2  != 3, 4
 738        // #3  == 7, 7
 739        // #4  == 9, 9
 740        // ...
 741        // #10 == 51, 51
 742        // #11 == 53, 53
 743        // #12 != 55, 53
 744        //
 745        const int ContextWindowCount = 8;
 746        const int HalfContextWindowCount = ContextWindowCount >> 1;
 747        const int MaskHalfContextWindowCount = HalfContextWindowCount - 1;
 748        const int MaxNonEqualItemCount = 64;

 751        var beforeNonEqualCount = 0;

 754        var afterNonEqualReverseCount = 0;

 757        // The slots will be rotated by overriding the `a` again, when the `d` is reached, then the `b`, etc.
 758        ItemsCompared<T> a = default, b = default, c = default, d = default;
 759        SmallList<ItemsCompared<T>, Stack4<ItemsCompared<T>>> collectedItems = default;

 761        var nonEqualItemCount = 0;
 762        var collectedMaxNonEqualItems = false;

 777            {
 778                var exp = expectedEnumerator.Current;
 779                var act = actualEnumerator.Current;
 780                if (!Equals(exp, act))
 781                {

 783                    collectedMaxNonEqualItems = nonEqualItemCount > MaxNonEqualItemCount;
 784                    if (!collectedMaxNonEqualItems)
 785                    {

 787                        if (beforeNonEqualCount != 0)

 797                            }
